function get_user_all_data_with_status_count()
{
	
	if (isset($_SESSION['user_data'][1]['status'])) {
        $logged_in_user_type = $_SESSION['user_data'][1]['status'];

        // User side chat
        if ($logged_in_user_type == "User") {
            $query = "SELECT user_id, user_name, user_profile, user_login_status, (SELECT COUNT(*) FROM chat_message WHERE to_user_id = :user_id AND from_user_id = chat_user_table.user_id AND status = 'No') AS count_status
            FROM chat_user_table
            WHERE user_type = 'User'";
		// Admin side chat
        } elseif ($logged_in_user_type == "Admin") {
            $query = "SELECT user_id, user_name, user_profile, user_login_status, (SELECT COUNT(*) FROM chat_message WHERE to_user_id = :user_id AND from_user_id = chat_user_table.user_id AND status = 'No') AS count_status
            FROM chat_user_table
            WHERE user_type = 'User'";
        } else {
            echo "Error: Unexpected user type.";
        }

        $statement = $this->connect->prepare($query);
        $statement->bindParam(':user_id', $this->user_id);
        $statement->execute();
        $data = $statement->fetchAll(PDO::FETCH_ASSOC);
        return $data;
    } else {
        // Handle the case where 'status' is not set in the user_data array
        echo "Error: User type not found. Please log in.";
    }
}
